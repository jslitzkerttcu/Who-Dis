    ### 2. `app/static/js/clipboard.js` (≤40 lines)

    ```javascript
    /* Phase 6 SRCH-01: WYSIWYG clipboard serializer.
     *
     * Reads the currently-rendered profile card DOM, formats plain text per D-13
     * (key:value lines, blank line between sections), copies via the
     * Clipboard API, and surfaces success/failure through showToast().
     *
     * Unexpanded sections (aria-expanded="false") emit one "Not loaded" line.
     */
    (function () {
      'use strict';

      function serializeCard(cardEl) {
        const lines = [];

        // 1. Default-card fields (everything tagged with data-copy-field that is NOT inside a section body).
        cardEl.querySelectorAll('[data-copy-field]').forEach(function (el) {
          if (el.closest('[role="region"]')) return;  // section bodies handled below
          lines.push(el.dataset.copyField + ': ' + el.textContent.trim().replace(/\s+/g, ' '));
        });

        // 2. Each enrichment section.
        cardEl.querySelectorAll('button[aria-controls]').forEach(function (header) {
          const sectionTitle = header.textContent.trim().replace(/\s+/g, ' ');
          const expanded = header.getAttribute('aria-expanded') === 'true';
          const bodyId = header.getAttribute('aria-controls');
          const body = document.getElementById(bodyId);

          lines.push('');  // blank line between sections
          lines.push(sectionTitle);

          if (!expanded || !body) {
            lines.push('Not loaded');
            return;
          }
          const fields = body.querySelectorAll('[data-copy-field]');
          if (fields.length === 0) {
            lines.push('No data');
            return;
          }
          fields.forEach(function (el) {
            lines.push(el.dataset.copyField + ': ' + el.textContent.trim().replace(/\s+/g, ' '));
          });
        });

        return lines.join('\n');
      }

      window.copyProfileToClipboard = function (cardEl) {
        if (!cardEl) {
          showToast("Couldn't copy. Select the text and copy manually.", 'error');
          return;
        }
        const text = serializeCard(cardEl);
        if (!navigator.clipboard || !navigator.clipboard.writeText) {
          showToast("Couldn't copy. Select the text and copy manually.", 'error');
          return;
        }
        navigator.clipboard.writeText(text).then(
          function () { showToast('Copied profile to clipboard', 'success'); },
          function () { showToast("Couldn't copy. Select the text and copy manually.", 'error'); }
        );
      };
    })();
    ```

    ```python
    import csv
    import io
    import re
    from datetime import datetime, timezone

    @search_bp.route("/api/profile/<user_id>/export.csv")
    @require_role("viewer")
    @handle_errors(json_response=True)
    def profile_export_csv(user_id):
        """SRCH-02: Per-profile CSV export with source-attribution column.

        WYSIWYG (D-12): reads from the EmployeeProfile cache only. No live Graph
        or Genesys calls during export. Sections without cached data emit
        'Not loaded' rows tagged '[<Source>, not loaded]' in the source column.
        """
        from app.models.employee_profiles import EmployeeProfile  # match Plan 03 import style
        profile = EmployeeProfile.get_by_user_id(user_id)  # or actual method per Plan 03 SUMMARY
        raw = (profile.raw_data if profile else {}) or {}

        rows = []  # list[(field, value, source)]

        # Default-card fields — always present from EmployeeProfile cache.
        rows.append(("Name", raw.get("displayName") or "", "[Graph]"))
        rows.append(("Title", raw.get("jobTitle") or "", "[Graph]"))
        rows.append(("Department", raw.get("department") or "", "[Graph]"))
        manager = raw.get("manager")
        manager_name = manager.get("displayName") if isinstance(manager, dict) else (manager or "")
        rows.append(("Manager", manager_name, "[Graph]"))
        rows.append(("Email", raw.get("mail") or raw.get("userPrincipalName") or "", "[Graph]"))
        rows.append(("Phone", raw.get("mobilePhone") or raw.get("businessPhones", [None])[0] or "", "[Graph]"))
        rows.append(("Employee ID", raw.get("employeeId") or "", "[Graph]"))

        # M365 enriched section — readable from cache (signInActivity, assignedLicenses).
        sign_in = (raw.get("signInActivity") or {}).get("lastSignInDateTime")
        if sign_in:
            rows.append(("Last sign-in", sign_in, "[Graph]"))
        else:
            rows.append(("Last sign-in", "Not loaded", "[Graph, not loaded]"))

        # Licenses — resolve via sku_catalog; if no assignedLicenses cached, "Not loaded".
        sku_catalog = current_app.container.get("sku_catalog")
        licenses = raw.get("assignedLicenses")
        if licenses:
            license_names = [
                sku_catalog.get_sku_name(l.get("skuId")) or l.get("skuId", "")
                for l in licenses if l.get("skuId")
            ]
            rows.append(("Licenses", "; ".join(filter(None, license_names)) or "(none)", "[Graph]"))
        else:
            rows.append(("Licenses", "Not loaded", "[Graph, not loaded]"))

        # MFA — D-02 says lazy / live only. WYSIWYG export never has it cached.
        rows.append(("MFA", "Not loaded", "[Graph, not loaded]"))

        # Genesys — also lazy. WYSIWYG export emits "Not loaded".
        rows.append(("Genesys queues", "Not loaded", "[Genesys, not loaded]"))
        rows.append(("Genesys skills", "Not loaded", "[Genesys, not loaded]"))
        rows.append(("Genesys presence", "Not loaded", "[Genesys, not loaded]"))

        # Build CSV body.
        buf = io.StringIO()
        writer = csv.writer(buf, quoting=csv.QUOTE_MINIMAL)
        writer.writerow(["Field", "Value", "Source"])
        writer.writerows(rows)

        # Sanitize username for filename — UI-SPEC line 117.
        username_seed = (raw.get("userPrincipalName") or raw.get("mail") or user_id).split("@")[0]
        username = re.sub(r"[^A-Za-z0-9_.-]", "_", username_seed) or "user"
        yyyymmdd = datetime.now(timezone.utc).strftime("%Y%m%d")
        filename = f"whodis-{username}-{yyyymmdd}.csv"

        # Audit per D-15.
        audit_service = current_app.container.get("audit_logger")
        user_email = getattr(g, "user", "unknown")
        audit_service.log_search(
            user_email=user_email,
            search_query=f"profile_export:{user_id}",
            results_count=len(rows),
            services=["Graph", "Genesys"],
            ip_address=request.headers.get("X-Forwarded-For", request.remote_addr),
            user_agent=request.headers.get("User-Agent"),
            success=True,
        )

        resp = make_response(buf.getvalue())
        resp.headers["Content-Type"] = "text/csv; charset=utf-8"
        resp.headers["Content-Disposition"] = f'attachment; filename="{filename}"'
        return resp
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser ↔ Flask (CSV download) | Authenticated GET; user-supplied `<user_id>` and the requesting user's email both flow into the audit trail. |
| Browser DOM ↔ Clipboard API | Plain-text payload contains potentially-sensitive values (department, manager, license names, MFA methods if expanded). |
| Server filename construction | `userPrincipalName` from cache flows into a Content-Disposition filename. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-16 | Information Disclosure | CSV export contains license + sign-in info | mitigate | `@require_role("viewer")` enforces minimum role. Audit-log per D-15. No data is exposed beyond what the user could already see in the UI. |
| T-06-17 | Tampering | `userPrincipalName` interpolated into Content-Disposition filename | mitigate | Sanitization regex `re.sub(r"[^A-Za-z0-9_.-]", "_", seed)` strips anything that could break the header (CRLF, quotes, semicolons). Falls back to `"user"` if the result is empty. |
| T-06-18 | Information Disclosure | Plain-text clipboard payload stored in OS clipboard | accept | User explicitly invoked the action; data is what they were already viewing. Browser's native clipboard scope is OS-level — out of WhoDis's enforcement boundary. |
| T-06-19 | XSS | clipboard.js reads `textContent` (not `innerHTML`) | mitigate | `textContent` is safe — never interprets HTML. The serializer never injects DOM, only reads. |
